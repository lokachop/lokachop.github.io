-- Gravel texture procedurally generated
ZVox.NewTexturePixelFunc("gravel", function(x, y)
	local spx = ZVox.Simplex2D(x, y)
	spx = (spx + 1) * .5

	local val = 64 + (spx * 140)
	local valR = val + (math.random() * 16)

	return valR, val, val
end)
